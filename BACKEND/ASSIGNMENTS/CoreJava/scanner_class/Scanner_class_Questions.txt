        while(scanner.hasNext()) {
            if(scanner.hasNextInt()) {
                System.out.println(scanner.nextInt());
            } else if(scanner.hasNextDouble()) {
                System.out.println(scanner.nextDouble());

        }
        scanner.close();
    }
}
